\section*{Notation and acronyms}
\vspace{-0.3em}

\subsection*{Symbols}
\noindent\begin{tabularx}{\linewidth}{@{}p{0.18\linewidth}X@{}}
\textbf{Symbol} & \textbf{Meaning} \\ \hline
\(P\) & Polynomial order per element (spectral/\(hp\) order). \\
\(h\) & Element size (mesh resolution). \\
\(hp\) & Adaptive strategy combining \(h\)-refinement and \(P\)-refinement. \\
\(t_{\text{step}}\) & Wall-time per timestep (e.g., ms/step). \\
\(\rho\) & Imbalance ratio \(=\frac{\max(\text{rank time/load})}{\text{mean}(\text{rank time/load})}\) (ideal \(\rho=1\)). \\
\(\eta\) & Effective efficiency \(\approx \frac{\text{mean}}{\max}\) (ideal \(\eta\approx 1\)). \\
\(k\) & Rebalance cadence in periodic policies (“rebalance every \(k\)”). \\
\(\Delta T_{\text{benefit}}\) & Predicted wall-time reduction from rebalancing (model-based). \\
\(\Delta T_{\text{cost}}\) & Predicted repartition+migration+coordination cost (model-based). \\
\end{tabularx}
